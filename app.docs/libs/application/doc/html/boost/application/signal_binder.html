Class signal_binder

boost::application::signal_binder — This class is the base of signal_manager class. User can extend of signal_manager to customize SIGANAL/Handlers of your application.

Synopsis

// In header: <boost/application/signal_binder.hpp>


class signal_binder {
public:
  // construct/copy/destruct
  explicit signal_binder(context &);
  explicit signal_binder(global_context_ptr);
  ~signal_binder();

  // public member functions
  void bind(int, const handler<> &);
  void bind(int, const handler<> &, const handler<> &);
  void bind(int, const handler<> &, boost::system::error_code &);

            boost::system::error_code &);
  void unbind(int);
  void unbind(int, boost::system::error_code &);

  // protected member functions
